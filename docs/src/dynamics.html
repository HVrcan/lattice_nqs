113    def _geometric_tdvp(self, parameters):
114        """
115        Calculates the derivative using the geometric formulation.
116
117        1. Transform the S-matrix and F-vector into real-like parametrization.
118        2. Calculate the geometric elements: metric, symplectic form.
119        3. Formulate and solve the AX=B equation using the `np.linalg.lstsq` convergent solver.
120        4. Get rid of the Langrange multipliers and repatch back into complex-like parametrization.
121        """
122        # first update parameters
123        self.sampler.update(parameters)
124
125        # Quantum Fisher Matrix and the energy gradient
126        S = self.sampler.S
127        F = self.sampler.F
128        # and the Lnorm gradient
129        Ln = self.sampler.Lnorm_grad
130
131        # split them into real parametrization
132        S = np.kron(S,[[1,1j],[-1j,1]])
133        F = np.kron(F,[1.,-1j])
134        Ln = np.kron(np.conjugate(Ln), [1,1j])
135        Ln = np.pad(Ln, (0, len(Ln)), mode='constant', constant_values=0)
136        metric = 2.*np.real(S)
137        symform = 2.*np.imag(S)
138
139        # A X = B problem
140        B = np.kron([0,1],F)
141        A = np.kron([[1,0],[0,0]],2.*metric) + np.kron([[0,1],[0,0]],symform.T) + np.kron([[0,0],[1,0]],symform)
142        #add the norm gradient
143        if self._lagrange_norm:
144            A = np.column_stack((A, Ln))
145            A = np.vstack((A, np.append(Ln, 0)))
146            B = np.append(B, 0.)
147        A = A.astype(complex)
148
149
150        # calculate the derivative
151        der = np.linalg.lstsq(A, -B, rcond = 1e-10)[0]
152
153        # tame if taming is on
154        if self._taming:
155            der =  der / (1+self.dt*np.abs(der))
156            #der =  der / (1+self.dt*np.linalg.norm(der))
157        
158        # return to the complex parametrization
159        try: # try to get rid of the Lagrange multipliers
160            if self._lagrange_norm:
161                der = der[:-1] # to get rid of the multiplier for norm conservtion
162            der = der[:int(len(der)/2)] # lambdas
163        except: # it's only problematic if the dimensions of the array are somehow wrong (but the code is quite proof to this)
164            print("Dimension of array must be a multiple of 4. Dimension is {}.".format(len(der)))
165        
166        der = der[::2] + 1j*der[1::2] # get back into the complex
167
168        return der

184    def _diagonal_tdvp(self, parameters, criterion = 1e-8):
185        """
186        Calculates the derivative using diagonalization.
187        1. diagonalizes the S-matrix,
188        2. transforms everything into the diagonal basis,
189        3. removes zero eigenvalues and constituend components,
190        4. updates without the nullspace,
191        5. transforms back.
192        """
193        # update the sampler
194        self.sampler.update(parameters)
195
196        # diagonalize the S matrix
197        evals, evecs = np.linalg.eig(self.sampler.S)
198        idx = np.argsort(np.abs(evals)) # instruction how to order them according to the eigenvalues
199        evals = evals[idx] #sorts them by those indices
200        evecs = evecs[:,idx] #also, but since it's a matrix and eigenvectors are collumns, sort by second index
201        evecs_inv = np.linalg.inv(evecs)
202        slicer = np.searchsorted(np.abs(evals), criterion, side='right')
203
204        # transform everything
205        #Sdiag = np.diag(evals)
206        Sdiag = (evecs_inv @ self.sampler.S) @ evecs
207        Fdiag = evecs_inv @ self.sampler.F
208
209        # slice things
210        Sdiag = Sdiag[slicer:, slicer:]
211        Fdiag = Fdiag[slicer:]
212
213        # now calculate the update
214        # it's just the TDVP update on the nonzero-subspace
215        der = -1j * np.linalg.inv(Sdiag) @ Fdiag
216
217        #fill it with zeros to match the length
218        der = np.pad(der, (len(parameters) - len(der), 0))
219        #transform it back to the original basis and return
220        der = evecs@der
221
222        if self._taming:
223            return der / (1+self.dt*np.abs(der))
224            #return der / (1+self.dt*np.linalg.norm(der))
225        else:
226            return der

240    def integrate(self, state):
241        """
242        Predicts the parameter vector in the next step of the time integration, from the initial `state`, using the selected integrator.\\
243        Integrator choices:
244        - `"heun"`: 2nd order explicit (RK2)
245        - `"implicit_midpoint"`: Newton-Rhapson implicit method, solution obtained by the `sp.optimize.newton` root finder
246
247        NOTE: taming is governed by the class property and affects the calculation of derivatives
248        """
249        if self._integrator == 'heun':
250            der = self.derivative(state) # euler step
251            wtilda = state + self.dt*der
252            dertilda = self.derivative(wtilda) # corrector step
253            der = 0.5 * (der + dertilda) # derivative in total
254            newstate = state + self.dt*der
255
256        elif self._integrator == 'implicit_midpoint':
257            implicit_equation = lambda wn: wn - state - self.dt*self.derivative(0.5*(wn + state))
258            #newstate = sp.optimize.newton(implicit_equation, state, maxiter = 10000, tol = 1e-10)
259            newstate = sp.optimize.newton(implicit_equation, (np.random.randn(len(state)) + 1j*np.random.randn(len(state)))*0.1, maxiter = 10000, tol = 1e-10)
260        
261        else:
262            print("Invalid integration method given.")
263            return
264
265        return newstate

269        """
270        Evolve the system in time using the TDVP EoM.
271
272        Steps:
273        1. Perturb the Hamiltonian.
274        2. Calculate the next parameter preiction.
275        3. Pile up the states and the time-dependent observables (energy)
276        """
277
278        w = self.states[-1].copy() # initial array
279
280        print("running simmulation...")
281        with tqdm(total=self.steps) as pbar:
282            for s in range(self.steps):
283                time = self._start + s*self.dt # hoe laat is het?
284                # perturb the Hamiltonian at that time
285                self.sampler.hspace.perturb(self.perturbator(time))
286
287                wnew = self.integrate(w.copy()) # calculate the new state
288                
289                # append metric things before update
290                self.residuals.append(self.sampler.residual(w, (wnew - w)/self.dt)) # residual distance
291                self.metric_velocity.append(self.sampler.metric_velocity(w, (wnew - w)/self.dt)) # the integrand in the metric distance
292                self.arc.append(np.sum(self.metric_velocity) * self.dt) # travelled arc length
293                self.gradient_velocity.append(self.sampler.gradient_velocity_rt(w, (wnew - w)/self.dt))
294                self.path.append(np.sum(self.gradient_velocity) * self.dt)
295                
296                # update the sampler with new parameters
297                self.sampler.update(wnew)

299                # append things after update

524    def gradient_descent_with_convergence(self, ref_state, 
525                                        initial_params, 
526                                        learning_rate=0.01, threshold=1e-6, miniter = 100, maxiter=1000,
527                                        learning_rate_decay=1.,
528                                        artificial_noise = 1e-4,
529                                        verbose = False):
530        """Performs a gradient descent algorithm with convergence criterion for stoping.
531
532        Variables
533        -------------------
534        - `ref_state`: state you want to fit your wavefunction to
535        - `initial_params`: starting parameter array
536
537        Kwargs
538        ----------------
539        - `learning_rate`: default is 0.01
540        - `threshold`: convergence criterion for the descent, default is 1e-6 (loss function)
541        - `miniter`, `maxiter`: hard limitations for iterations
542        - `learning_rate_decay`: if supplied, scales down the learning rate every iteration, default values 1 means no scaling
543        - `artificial_noise`: add Gaussian noise with this much intensity into the gradient
544        - `distance_target`: measure the straight-line distance towards this target, by default `None` so it doesn't measure it
545
546        Returns the evolution of parameters, the loss function, the psi norm, metric velocities, travelled arc, residual distances, and optionally straight-line distances to the provided target.
547        """
548        
549        params = np.array(initial_params, dtype=complex)
550        iteration = 0
551
552        #accumulated values
553        loss_values = [self.wavefunction_infidelity(ref_state, params)]
554        param_list = [params.copy()]
555        norm_values = []
556        
557
558        while iteration < maxiter:
559            gradient = self.analytical_infidelity_gradient(params, ref_state) # calculate gradient
560            gradient = full.add_noise(gradient, eta = artificial_noise) # add artifficial noise to the gradient to jostle it out of a local minimum
561            invS = np.linalg.pinv(self.sampler.S + 1.e-4*np.eye(len(self.sampler.S))) # calculate the inverse of the S matrix
562
563            # update the parameters
564            params -= learning_rate * invS@gradient # update the parameters
565
566            # append stuff
567            current_loss = self.wavefunction_infidelity(ref_state, params)
568            loss_values.append(current_loss)
569            param_list.append(params.copy())
570            norm_values.append(np.linalg.norm(self.sampler.psiket))
571
572            if iteration >= miniter and np.abs(current_loss) < threshold:
573                break
574
575            iteration += 1
576

114    def _geometric_tdvp(self, parameters):
115        """
116        Calculates the derivative using the geometric formulation.
117
118        1. Transform the S-matrix and F-vector into real-like parametrization.
119        2. Calculate the geometric elements: metric, symplectic form.
120        3. Formulate and solve the AX=B equation using the `np.linalg.lstsq` convergent solver.
121        4. Get rid of the Langrange multipliers and repatch back into complex-like parametrization.
122        """
123        # first update parameters
124        self.sampler.update(parameters)
125
126        # Quantum Fisher Matrix and the energy gradient
127        S = self.sampler.S
128        F = self.sampler.F
129        # and the Lnorm gradient
130        Ln = self.sampler.Lnorm_grad
131
132        # split them into real parametrization
133        S = np.kron(S,[[1,1j],[-1j,1]])
134        F = np.kron(F,[1.,-1j])
135        Ln = np.kron(np.conjugate(Ln), [1,1j])
136        Ln = np.pad(Ln, (0, len(Ln)), mode='constant', constant_values=0)
137        metric = 2.*np.real(S)
138        symform = 2.*np.imag(S)
139
140        # A X = B problem
141        B = np.kron([0,1],F)
142        A = np.kron([[1,0],[0,0]],2.*metric) + np.kron([[0,1],[0,0]],symform.T) + np.kron([[0,0],[1,0]],symform)
143        #add the norm gradient
144        if self._lagrange_norm:
145            A = np.column_stack((A, Ln))
146            A = np.vstack((A, np.append(Ln, 0)))
147            B = np.append(B, 0.)
148        A = A.astype(complex)
149
150
151        # calculate the derivative
152        der = np.linalg.lstsq(A, -B, rcond = 1e-10)[0]
153
154        # tame if taming is on
155        if self._taming:
156            der =  der / (1+self.dt*np.abs(der))
157            #der =  der / (1+self.dt*np.linalg.norm(der))
158        
159        # return to the complex parametrization
160        try: # try to get rid of the Lagrange multipliers
161            if self._lagrange_norm:
162                der = der[:-1] # to get rid of the multiplier for norm conservtion
163            der = der[:int(len(der)/2)] # lambdas
164        except: # it's only problematic if the dimensions of the array are somehow wrong (but the code is quite proof to this)
165            print("Dimension of array must be a multiple of 4. Dimension is {}.".format(len(der)))
166        
167        der = der[::2] + 1j*der[1::2] # get back into the complex
168
169        return der

185    def _diagonal_tdvp(self, parameters, criterion = 1e-8):
186        """
187        Calculates the derivative using diagonalization.
188        1. diagonalizes the S-matrix,
189        2. transforms everything into the diagonal basis,
190        3. removes zero eigenvalues and constituend components,
191        4. updates without the nullspace,
192        5. transforms back.
193        """
194        # update the sampler
195        self.sampler.update(parameters)
196
197        # diagonalize the S matrix
198        evals, evecs = np.linalg.eig(self.sampler.S)
199        idx = np.argsort(np.abs(evals)) # instruction how to order them according to the eigenvalues
200        evals = evals[idx] #sorts them by those indices
201        evecs = evecs[:,idx] #also, but since it's a matrix and eigenvectors are collumns, sort by second index
202        evecs_inv = np.linalg.inv(evecs)
203        slicer = np.searchsorted(np.abs(evals), criterion, side='right')
204
205        # transform everything
206        #Sdiag = np.diag(evals)
207        Sdiag = (evecs_inv @ self.sampler.S) @ evecs
208        Fdiag = evecs_inv @ self.sampler.F
209
210        # slice things
211        Sdiag = Sdiag[slicer:, slicer:]
212        Fdiag = Fdiag[slicer:]
213
214        # now calculate the update
215        # it's just the TDVP update on the nonzero-subspace
216        der = -1j * np.linalg.inv(Sdiag) @ Fdiag
217
218        #fill it with zeros to match the length
219        der = np.pad(der, (len(parameters) - len(der), 0))
220        #transform it back to the original basis and return
221        der = evecs@der
222
223        if self._taming:
224            return der / (1+self.dt*np.abs(der))
225            #return der / (1+self.dt*np.linalg.norm(der))
