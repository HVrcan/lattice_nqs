  6def add_noise(f, eta = 0.):
  7    """
  8    Adds a noise factor to the vector. The noise added to each component `fi` is a complex number drawn from a normal distribution with:\\
  9    `variance = |eta*fi|^2`.
 10    - `eta`: noise strength factor (typically small).
 11
 12    Returns the noisified vector.
 13    """
 14    variances = np.abs(f*eta)**2. # array of variances
 15
 16    # make the noisy vector
 17    noise_r = np.random.normal(scale=np.sqrt(variances))
 18    noise_i = np.random.normal(scale=np.sqrt(variances))
 19    noise = noise_r + 1j*noise_i
 20
 21    return f + noise

340    def residual(self, w, wdot):
341        """
342        Calculates the residual distance.
343        
344        - `w`: current parameters
345        - `wdot`: derivative of the parameter vector
346
347        Careful with indices - this thing is calculared from the update directly, so it will always have one less values.
348        """
349
350        # residual distance is the function of time, so it must also come from S and F at THIS time
351        self.update(w)
352
353        res = np.vdot(wdot, self.S@wdot) # the S-matrix part
354        res += 1j * np.vdot(wdot, self.F) - 1j * np.vdot(self.F, wdot) # the F part
355        res += self.E2var() - self.Evar()**2 # the self energy part
356
357        #return np.real(res*(np.linalg.norm(self.psiket)**2))
358        return np.real(res) # get rid of the imaginary part

414    def line_distance(self, w, target_w, division = 100):
415        """
416        Measures the distance from point `w` to point `target_w` on the manifold along a straigh line trajectory. Still takes into account the S-matrix as metric.
417
418        - `division`: discretization of the line between the two points
419        """
420        params = w.copy()
421        v = target_w - params # constant tangent vector between the two poitns
422        dv = v/division # step along the tangent
423
424        # update the sampler
425        self.update(params)
426        d = np.sqrt( np.real(
427            np.vdot(v, self.S @ v)
428            ))
429
430        for _ in range(division):
431            params += dv # add small vector
432            self.update(params) # update the sampler
433
434            d += np.sqrt( np.real( # add small distance
435            np.vdot(v, self.S @ v)
436                ))
437
438        return d/division

230
231            #loop over the first parameter index

Handles the calculation of averages and variational manifold elements for the case of full sampling of the Hilbert space by RBM.

Variables

  • hspace: the Hilbert space object
  • alpha: network density
  • tinvariant: is the network translationally invariant? this changes the number of used parameters
  • normal: use normal full averaging? (as opposed to Monte-Carlo-like)
  • scale: multiply the wave function with this number
  • real: are the parameters real?
  • noise_level: controls how much artificial noise you introduce into the energy gradient

Properties

  • network: contains rbm elements (see rbm.py)
  • psiket: ket in the Hilbert (sub)space
  • S: quantum Fisher matrix
  • F: energy gradient
  • Lnorm: gradient of the norm loss

Methods

  • Evar: average energy
  • Cvar(link): average correlation at link
  • Czvar(link): average z-correlation at link
  • update(parameters): updates the network with new parameters and reflects the updates to the sampler
